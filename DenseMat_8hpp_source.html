287 void OCP_ABpC(const T1& m, const T1& n, const T1& k, const T2* A, const T2* B, T2* C) {
288  // C = AB + C
289  // A: m*k B:k*n C:m*n
290  // all matrix are row majored matrices
291  for (T1 i = 0; i < m; i++) {
292  for (T1 j = 0; j < n; j++) {
293  for (T1 l = 0; l < k; l++) {
294  C[i * n + j] += A[i * k + l] * B[l * n + j];
295  }
296  }
